% Denna fil kan användas som sådan, bara verserna,
% namnen och annan rådata behöver bytas ur fälten.
% Tecknet "%" markerar en kommentar som helt och 
% hållet ignoreras av programmet som läser filen.

% Spara den färdiga filen som 
% 'SangnamnUtanMellanslagEllerSkander.tex'
% t.ex. blir "Vid En Källa" till 
% 'VidEnKalla.tex'
% Varje sång blir en egen fil.

\beginsong{Om jag ska på fest}[ 	% Börja sången här
	by={Anders Nyström},	% Författare
	sr={Och flickan hon går i ringen}]		% Melodi
			% Alternativa
			% sångnamn
	
\beginverse*		% Börja vers
Jag äter fem kilo socker om jag ska på fest
och dricker en massa vatten som blandats med jäst.
Sen skakar jag kroppen tills allt blandat sej.
Tar nubbarna slut kan ni hälla ur mej!
\endverse			% Sluta vers

\endsong			% Sluta sång
