\beginsong{Punschen kommer}[				% Författare
	sr={Vals ur Glada änkan}]						% sångnamn
	

\beginverse*						% Börja vers
Punschen kommer,
punschen kommer
ljuv och sval.
Glasen imma,
röster stimma,
i vår sal.
Skål för glada minnen,
skål för varje vår.
Inga sorger finnes mer när punsch vi får!
\endverse							% Sluta vers

\endsong							% Sluta sång
